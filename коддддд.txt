function applyTranslations() {
  document.querySelectorAll('[data-i18n]').forEach(el => {
    const key = el.getAttribute('data-i18n');
    if(el.tagName==='INPUT') el.placeholder = t(key);
    else el.innerHTML = t(key);
  });
}

const steps = [
  {key:'personal',icon:'👤'},
  {key:'education',icon:'🎓'},
  {key:'experience',icon:'💼'},
  {key:'skills',icon:'⚡'},
  {key:'achievements',icon:'🏆'},
  {key:'projects',icon:'🚀'},
  {key:'social',icon:'🔗'},
  {key:'extra',icon:'✨'}
];

function renderBuilder() {
  const stepsHtml = steps.map((s,i) => `
    <div class="step-item ${i===currentStep?'active':''} ${isStepDone(i)?'done':''}" onclick="goStep(${i})">
      <div class="step-num">${isStepDone(i)&&i!==currentStep?'✓':i+1}</div>
      <div class="step-label">${t('step.'+s.key)}</div>
    </div>`).join('');
  document.getElementById('steps-list').innerHTML = stepsHtml;
  const pct = Math.round(((currentStep+1)/steps.length)*100);
  document.getElementById('progress-bar').style.width = pct+'%';
  document.getElementById('progress-text').textContent = `${currentStep+1} / ${steps.length}`;
  renderStep(currentStep);
}

function isStepDone(i){
  const s = steps[i].key;
  if(s==='personal') return portfolio.personal.name.length>0;
  if(s==='education') return portfolio.education.length>0;
  if(s==='experience') return portfolio.experience.length>0;
  if(s==='skills') return portfolio.skills.length>0;
  if(s==='achievements') return portfolio.achievements.length>0;
  if(s==='projects') return portfolio.projects.length>0;
  if(s==='social') return portfolio.social.linkedin||portfolio.social.github;
  if(s==='extra') return portfolio.extra.hobbies.length>0;
  return false;
}

function goStep(i) { currentStep = i; renderBuilder(); }

function renderStep(step) {
  const s = steps[step];
  let html = '';
  if(s.key==='personal') html = renderPersonalStep();
  else if(s.key==='education') html = renderListStep('education','edu');
  else if(s.key==='experience') html = renderListStep('experience','exp');
  else if(s.key==='skills') html = renderSkillsStep();
  else if(s.key==='achievements') html = renderListStep('achievements','ach');
  else if(s.key==='projects') html = renderListStep('projects','proj');
  else if(s.key==='social') html = renderSocialStep();
  else if(s.key==='extra') html = renderExtraStep();
  document.getElementById('builder-content').innerHTML = html;
  // Attach live inputs
  attachPersonalListeners();
}

function uploadPhoto(input) {
  const file = input.files[0];
  if(!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    portfolio.personal.photo = e.target.result;
    const prev = document.getElementById('photo-preview');
    if(prev) prev.innerHTML = `<img src="${e.target.result}" style="width:100%;height:100%;object-fit:cover;">`;
    triggerAutoSave();
  };
  reader.readAsDataURL(file);
}

async function generateAboutMe() {
  const name = portfolio.personal.name || 'Professional';
  const prof = portfolio.personal.profession || 'professional';
  const btn = document.querySelector('.ai-panel button');
  const resultDiv = document.getElementById('ai-result');
  if(!resultDiv) return;
  resultDiv.style.display='block';
  resultDiv.innerHTML = `<div class="ai-loading"><div class="ai-dot"></div><div class="ai-dot"></div><div class="ai-dot"></div><span style="margin-left:4px;">${lang==='kk'?'Жасалуда...':lang==='ru'?'Генерация...':'Generating...'}</span></div>`;
  btn.disabled = true;
  try {
    const prompt = lang==='kk'
      ? `${name} атты ${prof} маманы үшін портфолиоға арналған кәсіби "Өзім туралы" бөлімін қазақша жаз. 3-4 сөйлем.`
      : lang==='ru'
      ? `Напиши профессиональный раздел "О себе" для портфолио специалиста по имени ${name}, работающего как ${prof}. 3-4 предложения, профессионально.`
      : `Write a professional "About Me" section for a portfolio of ${name}, a ${prof}. 3-4 sentences, confident and professional tone.`;
    const resp = await fetch('https://api.anthropic.com/v1/messages',{
      method:'POST',
      headers:{'Content-Type':'application/json'},
      body:JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:1000,messages:[{role:'user',content:prompt}]})
    });
    const data = await resp.json();
    const text = data.content?.map(c=>c.text||'').join('') || 'Error generating text.';
    resultDiv.innerHTML = `<div class="ai-result">${text}<br><button class="btn-primary btn-sm" style="margin-top:10px;" onclick="useAIText('${encodeURIComponent(text)}')">${lang==='kk'?'Пайдалану':lang==='ru'?'Использовать':'Use this text'}</button></div>`;
  } catch(e) {
    resultDiv.innerHTML = `<div class="ai-result" style="color:var(--gray-500);">${lang==='kk'?'Қате орын алды. Қайтадан көріңіз.':lang==='ru'?'Ошибка. Попробуйте снова.':'Error. Please try again.'}</div>`;

}

function saveModalEntry() {
  const type = modalType;
  const titleEl = document.getElementById('m-title');
  if(!titleEl || !titleEl.value.trim()) { showToast(lang==='kk'?'Атауды енгізіңіз':lang==='ru'?'Введите название':'Please enter a title'); return; }
  if(type==='skills') {
    const lvl = document.getElementById('m-level');
    portfolio.skills.push({name:titleEl.value.trim(),level:lvl?parseInt(lvl.value):75});
  } else if(type==='education') {
    portfolio.education.push({title:titleEl.value.trim(),subtitle:(document.getElementById('m-subtitle')||{}).value||'',year:(document.getElementById('m-year')||{}).value||''});
  } else if(type==='experience') {
    portfolio.experience.push({company:titleEl.value.trim(),position:(document.getElementById('m-subtitle')||{}).value||'',desc:(document.getElementById('m-desc')||{}).value||'',year:(document.getElementById('m-year')||{}).value||''});
  } else if(type==='achievements') {
    portfolio.achievements.push({award:titleEl.value.trim(),desc:(document.getElementById('m-desc')||{}).value||'',year:(document.getElementById('m-year')||{}).value||''});
  } else if(type==='projects') {
    portfolio.projects.push({name:titleEl.value.trim(),desc:(document.getElementById('m-desc')||{}).value||'',tech:(document.getElementById('m-tech')||{}).value||''});
  }
  closeModal();
  triggerAutoSave();
  renderBuilder();
}

function removeEntry(type, idx) {
  portfolio[type].splice(idx,1);
  triggerAutoSave();
  renderBuilder();
}

function renderPreview() {
  renderTemplatePreviewList();
  const p = portfolio.personal;
  const avatarHtml = p.photo
    ? `<img src="${p.photo}" style="width:100%;height:100%;object-fit:cover;">`
    : `<span>${(p.name||'?').charAt(0).toUpperCase()}</span>`;
  const templates = [renderModern,renderMinimal,renderCreative,renderTeacher,renderElegant];
  const html = (templates[selectedTemplate]||renderModern)(portfolio, avatarHtml);
  document.getElementById('portfolio-preview').innerHTML = html;
  // Quality score
  let score = 0;
  if(p.name) score+=15;
  if(p.profession) score+=10;
  if(p.about) score+=15;
  if(portfolio.education.length) score+=15;
  if(portfolio.experience.length) score+=15;
  if(portfolio.skills.length) score+=10;
  if(portfolio.achievements.length) score+=10;
  if(portfolio.social.linkedin||portfolio.social.github) score+=10;
  document.getElementById('quality-score').textContent = score+'%';
  document.getElementById('quality-label').textContent = score>=80?t('quality.excellent'):score>=50?t('quality.good'):t('quality.fair');
}
